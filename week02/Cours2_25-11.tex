\section{Le modèle à un locus}
% Slide 1 : décomposition phénotypique
\begin{frame}
  \frametitle{Décomposition de la valeur génétique (G)}

  \textbf{Population :} panmictique, taille infinie, reproduction sexuée.\\[0.15cm]
  \textbf{Un seul locus} $A$ influence la valeur génétique.\\
  Génotypes : $A_iA_j$ avec $A_iA_j = A_jA_i$.

  \vspace{0.3cm}
  \begin{itemize}
    \item<2-> $P$ : Phénotype
    \item<3-> $\mu$ : Moyenne de la population
    \item<4-> $G$ : Valeur génétique
    \item<5-> $E$ : Effet de l'environnement
  \end{itemize}

  \only<6->{
    \[
      P = \mu + G + E
    \]
  }

\end{frame}

% Slide 2 : modèle génétique de Fisher
\begin{frame}
  \frametitle{Décomposition de la valeur génétique (G)}

  Valeur génétique d'un génotype $A_iA_j$ :
  \[
    G_{ij} = \alpha_i + \alpha_j + \beta_{ij}
  \]

\[
\begin{array}{r l c}
  G = & \text{effet moyen du gène paternel} & + \\
      & \text{effet moyen du gène maternel} & + \\
      & \text{effet d'interaction entre (dominance)} & 
\end{array}
\]

  \vspace{0.3cm}
  où :
  \begin{itemize}
    \item<2-> $\alpha_i$ : effet additif de l’allèle $A_i$
    \item<3-> $\alpha_j$ : effet additif de l’allèle $A_j$
    \item<4-> $B_{ij}$ : déviation due à la dominance (interaction $A_iA_j$)
  \end{itemize}

\end{frame}


\begin{frame}
\frametitle{Effet moyen associé à un allèle (\(\alpha_i\))}

L'effet moyen de l'allèle \(A_i\) (average effect, Fisher) est défini comme :

\[
\alpha_i = E_j(G_{ij}) - \mu = \sum_j p_j G_{ij} - \mu
\]

\begin{itemize}
  \item \(p_j\) : fréquence de l'allèle \(A_j\)
  \item Panmixie : \(j\) est tiré indépendamment de \(i\)
  \item \(\alpha_i\) a par construction une \textbf{moyenne nulle} dans la population
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Effet de dominance (\(\beta_{ij}\))}

L'effet de dominance est défini comme l'interaction entre les deux allèles :

\[
\beta_{ij} = G_{ij} - \alpha_i - \alpha_j
\]

\begin{itemize}
  \item Moyenne nulle : \(E_j(\beta_{ij}) = 0\)
  \item Par construction, \(\alpha_i\) et \(\beta_{ij}\) sont \textbf{indépendants}
  \item Covariances nulles entre effets tirés indépendamment :
  \[
  \text{cov}(\alpha_i, \beta_{ij}) = 0 \quad 
  \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Indépendance des effets alléliques}

Sous panmixie, les deux allèles \(i\) et \(j\) d'un individu sont tirés indépendamment :

\[
\text{cov}(\alpha_i, \alpha_j) = 0
\]

De même : 
\[
E(\alpha_j) = E(\beta_{ij}) = 0
\]
\end{frame}

\begin{frame}
\frametitle{Variance génétique}

La variance génétique d'un génotype est :

\[
V_G = V(G_{ij}) = V(\alpha_i) + V(\alpha_j) + V(\beta_{ij}) \quad (\text{covariances nulles})
\]

\[
V_G = 2 V(\alpha) + V(\beta)
\]

On définit :
\[
\begin{aligned}
V_A &= 2 V(\alpha) = 2 E(\alpha_i^2) \quad \text{(variance additive)}\\
V_D &= V(\beta) = E(\beta_{ij}^2) \quad \text{(variance de dominance)}
\end{aligned}
\]

Donc :
\[
V_G = V_A + V_D
\]
\end{frame}


\begin{frame}
\frametitle{Héritabilité au sens strict}

L'héritabilité au sens strict (\(h^2_{ss}\)) est la proportion de la variance phénotypique due à la \textbf{variance génétique additive} :

\[
h^2_{ss} = \frac{V_A}{V_P}
\]

où \(V_P\) est la variance phénotypique totale.
\end{frame}

\section{Généralisation du modèle}
\begin{frame}
\frametitle{Généralisation à plusieurs locus}

\[
G_{l_{ij}} = \alpha_{l_i} + \alpha_{l_j} + \beta_{l_{ij}}
\]

\begin{itemize}
  \item \(\alpha_{l_i}\) : effet additif de l’allèle \(i\) au locus \(l\)  
  \item \(\alpha_{l_j}\) : effet additif de l’allèle \(j\) au locus \(l\)  
  \item \(\beta_{l_{ij}}\) : déviation due à la dominance\(ij\) au locus \(l\)
\end{itemize}

\end{frame}



\begin{frame}
\frametitle{Généralisation à plusieurs locus}
\begin{itemize}
  \item En l'absence d'épistasie et pour des loci en équilibre de liaison : 
  \textbf{les variances s'additionnent} pour un caractère quantitatif.
  \item Variance additive totale : 
  \[
    V_A = 2 \sum_{\text{locus}} V(\alpha_{\text{loc}}) = 2 \sum_{\text{locus}} E(\alpha_{i,\text{loc}}^2)
  \]
  \item Variance de dominance totale : 
  \[
    V_D = \sum_{\text{locus}} V(\beta_{\text{loc}}) = \sum_{\text{locus}} E(\beta_{ij,\text{loc}}^2)
  \]
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Signification génétique des effets additifs}

\begin{itemize}
  \item \textbf{Effet moyen d'un allèle} : valeur moyenne d'un allèle en combinaison avec tous les autres.
  \item Dépend de l'effet biologique propre et des \textbf{fréquences alléliques}.
  \item Les variances \textbf{diffèrent selon la population} et au cours du temps.
  \item On peut utiliser un \textbf{modèle biologique} pour fixer les contributions des allèles/génotypes.
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Cas d'une population biallélique (Fisher)}

\begin{itemize}
  \item Population panmictique, 1 locus, 2 allèles : B et b, fréquences \(p\) et \(q\)
  \item Trois génotypes possibles :
  \[
    BB: p^2,\quad Bb: 2pq,\quad bb: q^2
  \]
  \item Valeurs phénotypiques moyennes arbitraires mais fixes : \(Y_{BB}, Y_{Bb}, Y_{bb}\)
\end{itemize}

\end{frame}



\begin{frame}
\frametitle{Cas d'une population biallélique (Fisher)}

\[
c = \frac{Y_{BB}+Y_{bb}}{2},\quad 
a = \frac{Y_{BB}-Y_{bb}}{2},\quad
d = Y_{Bb}-c
\]

\begin{itemize}
  \item \(d\) : paramètre lié à la \textbf{dominance} de B sur b
  \item Degré de dominance : \(d/a\)
  \begin{itemize}
    \item \(d/a = 1\) : dominance complète
    \item \(0<d/a<1\) : dominance partielle
    \item \(d/a>1\) : superdominance
  \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Phénotype et valeur génétique}

\[
\begin{aligned}
Y_{BB} &= \mu + G_{BB} = \mu + 2\alpha_B + \beta_{BB} \\
Y_{Bb} &= \mu + G_{Bb} = \mu + \alpha_B + \alpha_b + \beta_{Bb} \\
Y_{bb} &= \mu + G_{bb} = \mu + 2\alpha_b + \beta_{bb}
\end{aligned}
\]

\[
\mu = p^2 Y_{BB} + 2pq Y_{Bb} + q^2 Y_{bb} = c + a(p-q) + 2 p q d
\]
\end{frame}

\begin{frame}
\frametitle{Effets moyens (\(\alpha_B, \alpha_b\))}

\[
\alpha_B = p Y_{BB} + q Y_{Bb} - \mu = q \alpha, \quad
\alpha_b = -p \alpha
\]

\begin{itemize}
  \item \textbf{Remarque :} L'effet moyen dépend des \textbf{fréquences alléliques}
  \item Même allèle, même effet biologique, peut avoir un \textbf{effet moyen différent} selon la population
\end{itemize}

\end{frame}



\begin{frame}
\frametitle{Termes de dominance (\(\beta_{ij}\))}
\[
\beta_{BB} = -2 q^2 d,\quad
\beta_{Bb} = 2 p q d,\quad
\beta_{bb} = -2 p^2 d
\]

\begin{itemize}
  \item Déviation entre valeurs \textbf{additives prédites} et \textbf{valeurs réelles}
  \item Mesure de la dominance dans le modèle génétique
\end{itemize}
\end{frame}



\begin{frame}
\frametitle{Régression géno-phénotypique}

\begin{itemize}
  \item Dose d'allèle B : 0 (bb), 1 (Bb), 2 (BB)
  \item Approximation additive : 
  \[
    \text{phénotype prédit} = 2 \alpha \times \text{dose}
  \]
  \item \textbf{Pente de la régression} = \(\alpha = \alpha_B - \alpha_b\)
  \item La droite de régression passe près des génotypes les plus fréquents
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Interprétation et implications}
\begin{itemize}
\setlength{\itemsep}{0.6em}
  \item Variance additive maximale lorsque les fréquences sont équilibrées
  \item Superdominance ou dominance complète : variance peut s'annuler
  \item Relation \textbf{modèle biologique vs modèle statistique} n’est pas triviale
  \item Importance pour la \textbf{cartographie génétique, QTL, et prédiction de valeur génomique}
\end{itemize}

\end{frame}

\section{Conséquences pour la pratique de l’amélioration génétique}
\begin{frame}
\frametitle{Conséquences pour la pratique de l’amélioration génétique}

\begin{itemize}
\setlength{\itemsep}{0.6em}
  \item<2-> \textbf{Sélection sur la valeur génétique additive} :  
    Rechercher les individus ayant la meilleure valeur additive

  \item<3-> \textbf{Sélection sur les effets d'interaction} :  
    Rechercher les couples dont la descendance présentera des effets de dominance ou d’interaction maximaux
    
  \item<4-> \textbf{Complémentarité des deux voies} :  
    Dans la pratique, ces deux approches se complètent. Exemple :  
    \begin{enumerate}
      \item Sélection des parents à forte valeur additive  
      \item Recherche des meilleurs couples/parents parmi les géniteurs retenus
    \end{enumerate}
\end{itemize}
\end{frame}
